21 double timeAddVectorsCPU(int n, int k) {
22  // adds two vectors of size n, k times, returns total duration
23  std::clock_t start;
24  double duration;
25 
26  int A[n], B[n], C[n];
27  for (int i=0; i<n; i++) {
28  A[i] = i;
29  B[i] = n-i;
30  C[i] = 0;
31  }
32 
33  start = std::clock();
34  for (int i=0; i<k; i++) {
35  for (int j=0; j<n; j++)
36  C[j] = A[j] + B[j];
37  }
38 
39  duration = (std::clock() - start) / (double) CLOCKS_PER_SEC;
40  return duration;
41 }
